## Problem 111: Unique Paths
**LeetCode #62** | Count paths from top-left to bottom-right (right/down moves only).

### Brute Force -- O(2^(M+N)) recursion
```cpp
int uniquePaths_r(int m,int n){if(m==1||n==1)return 1;return uniquePaths_r(m-1,n)+uniquePaths_r(m,n-1);}
```

### Optimal -- O(M*N) DP (or O(min(M,N)) with math: C(m+n-2, m-1))
```cpp
int uniquePaths(int m, int n) {
    vector<vector<int>> dp(m,vector<int>(n,1));
    for(int r=1;r<m;r++) for(int c=1;c<n;c++) dp[r][c]=dp[r-1][c]+dp[r][c-1];
    return dp[m-1][n-1];
}
```

## Problem 112: Longest Common Subsequence
**LeetCode #1143**

### Brute Force -- O(2^(M+N)) recursion
### Optimal -- O(M*N) DP
```cpp
int longestCommonSubsequence(string a, string b) {
    int m=a.size(),n=b.size(); vector<vector<int>> dp(m+1,vector<int>(n+1,0));
    for(int i=1;i<=m;i++) for(int j=1;j<=n;j++)
        dp[i][j]=(a[i-1]==b[j-1])?dp[i-1][j-1]+1:max(dp[i-1][j],dp[i][j-1]);
    return dp[m][n];
}
```

## Problem 113: Best Time to Buy and Sell Stock with Cooldown
**LeetCode #309** | Cooldown of 1 day after selling.

### Brute Force -- O(2^N) try all buy/sell sequences
### Optimal -- O(N) state machine DP
```cpp
int maxProfit(vector<int>& prices) {
    int hold=INT_MIN,sold=0,rest=0;
    for(int p:prices){
        int ph=hold,ps=sold,pr=rest;
        hold=max(ph,pr-p);   // buy today or stay holding
        sold=ph+p;           // sell today
        rest=max(pr,ps);     // rest or coming off cooldown
    }
    return max(sold,rest);
}
```

## Problem 114: Coin Change II
**LeetCode #518** | Count combinations to make amount (unlimited use).

### Brute Force -- O(amount^coins) recursion
### Optimal -- O(coins * amount) unbounded knapsack DP
```cpp
int change(int amount, vector<int>& coins) {
    vector<int> dp(amount+1,0); dp[0]=1;
    for(int c:coins) for(int i=c;i<=amount;i++) dp[i]+=dp[i-c];
    return dp[amount];
}
```

## Problem 115: Target Sum
**LeetCode #494** | Assign + or - to each number; count ways to reach target.

### Brute Force -- O(2^N) try all + and - assignments
```cpp
int findTargetSumWays_r(vector<int>& n,int i,int t){if(i==n.size())return t==0;return findTargetSumWays_r(n,i+1,t-n[i])+findTargetSumWays_r(n,i+1,t+n[i]);}
int findTargetSumWays(vector<int>& nums,int target){return findTargetSumWays_r(nums,0,target);}
```

### Optimal -- O(N * sum) DP with offset
```cpp
int findTargetSumWays(vector<int>& nums, int target) {
    int sum=accumulate(nums.begin(),nums.end(),0);
    if(abs(target)>sum) return 0;
    unordered_map<int,int> dp; dp[0]=1;
    for(int x:nums){
        unordered_map<int,int> next;
        for(auto&[s,c]:dp){next[s+x]+=c;next[s-x]+=c;}
        dp=next;
    }
    return dp.count(target)?dp[target]:0;
}
```

## Problem 116: Interleaving String
**LeetCode #97** | Can s3 be formed by interleaving s1 and s2?

### Brute Force -- O(2^(M+N)) recursion
### Optimal -- O(M*N) DP
```cpp
bool isInterleave(string s1, string s2, string s3) {
    int m=s1.size(),n=s2.size();
    if(m+n!=(int)s3.size()) return false;
    vector<vector<bool>> dp(m+1,vector<bool>(n+1,false)); dp[0][0]=true;
    for(int i=1;i<=m;i++) dp[i][0]=dp[i-1][0]&&s1[i-1]==s3[i-1];
    for(int j=1;j<=n;j++) dp[0][j]=dp[0][j-1]&&s2[j-1]==s3[j-1];
    for(int i=1;i<=m;i++) for(int j=1;j<=n;j++)
        dp[i][j]=(dp[i-1][j]&&s1[i-1]==s3[i+j-1])||(dp[i][j-1]&&s2[j-1]==s3[i+j-1]);
    return dp[m][n];
}
```

## Problem 117: Longest Increasing Path in Matrix
**LeetCode #329**

### Brute Force -- O(M*N * 4^(M*N)) DFS from every cell without memoization
### Optimal -- O(M*N) DFS with memoization
```cpp
int longestIncreasingPath(vector<vector<int>>& mat) {
    int m=mat.size(),n=mat[0].size(),res=0;
    vector<vector<int>> memo(m,vector<int>(n,0));
    int dx[]={1,-1,0,0},dy[]={0,0,1,-1};
    function<int(int,int)> dfs=[&](int r,int c)->int{
        if(memo[r][c]) return memo[r][c];
        int best=1;
        for(int d=0;d<4;d++){int nr=r+dx[d],nc=c+dy[d];
            if(nr>=0&&nr<m&&nc>=0&&nc<n&&mat[nr][nc]>mat[r][c]) best=max(best,1+dfs(nr,nc));}
        return memo[r][c]=best;
    };
    for(int r=0;r<m;r++) for(int c=0;c<n;c++) res=max(res,dfs(r,c));
    return res;
}
```

## Problem 118: Distinct Subsequences
**LeetCode #115** | Count distinct ways s contains t as subsequence.

### Brute Force -- O(2^M) try all subsequences of s
### Optimal -- O(M*N) DP
```cpp
int numDistinct(string s, string t) {
    int m=s.size(),n=t.size(); vector<vector<long>> dp(m+1,vector<long>(n+1,0));
    for(int i=0;i<=m;i++) dp[i][0]=1;
    for(int i=1;i<=m;i++) for(int j=1;j<=n;j++){
        dp[i][j]=dp[i-1][j];
        if(s[i-1]==t[j-1]) dp[i][j]+=dp[i-1][j-1];
    }
    return dp[m][n];
}
```

## Problem 119: Edit Distance
**LeetCode #72** | Min operations (insert, delete, replace) to transform word1 to word2.

### Brute Force -- O(3^(M+N)) recursion
### Optimal -- O(M*N) DP
```cpp
int minDistance(string w1, string w2) {
    int m=w1.size(),n=w2.size(); vector<vector<int>> dp(m+1,vector<int>(n+1,0));
    for(int i=0;i<=m;i++) dp[i][0]=i; for(int j=0;j<=n;j++) dp[0][j]=j;
    for(int i=1;i<=m;i++) for(int j=1;j<=n;j++){
        if(w1[i-1]==w2[j-1]) dp[i][j]=dp[i-1][j-1];
        else dp[i][j]=1+min({dp[i-1][j],dp[i][j-1],dp[i-1][j-1]});
    }
    return dp[m][n];
}
```

## Problem 120: Burst Balloons
**LeetCode #312** | Interval DP: maximize coins from bursting balloons.

### Brute Force -- O(N! * N) try all permutations of bursting order
### Optimal -- O(N^3) interval DP (think: which balloon is burst LAST in interval)
```cpp
int maxCoins(vector<int>& nums) {
    nums.insert(nums.begin(),1); nums.push_back(1);
    int n=nums.size(); vector<vector<int>> dp(n,vector<int>(n,0));
    for(int len=2;len<n;len++){
        for(int l=0;l<n-len;l++){
            int r=l+len;
            for(int k=l+1;k<r;k++)
                dp[l][r]=max(dp[l][r],dp[l][k]+nums[l]*nums[k]*nums[r]+dp[k][r]);
        }
    }
    return dp[0][n-1];
}
```

## Problem 121: Regular Expression Matching
**LeetCode #10** | Match pattern with '.' (any char) and '*' (0+ of preceding).

### Brute Force -- O(2^(M+N)) recursion without memoization
### Optimal -- O(M*N) DP
```cpp
bool isMatch(string s, string p) {
    int m=s.size(),n=p.size(); vector<vector<bool>> dp(m+1,vector<bool>(n+1,false)); dp[0][0]=true;
    for(int j=2;j<=n;j++) if(p[j-1]=='*') dp[0][j]=dp[0][j-2];
    for(int i=1;i<=m;i++) for(int j=1;j<=n;j++){
        if(p[j-1]=='*'){
            dp[i][j]=dp[i][j-2]; // use '*' as 0 occurrences
            if(p[j-2]=='.'||p[j-2]==s[i-1]) dp[i][j]=dp[i][j]||dp[i-1][j];
        } else {
            dp[i][j]=(p[j-1]=='.'||p[j-1]==s[i-1])&&dp[i-1][j-1];
        }
    }
    return dp[m][n];
}
```

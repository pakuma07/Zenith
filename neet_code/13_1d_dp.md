## Problem 99: Climbing Stairs
**LeetCode #70** | N stairs, 1 or 2 steps at a time. Count ways.

### Brute Force -- O(2^N) recursion without memoization
```cpp
int climbStairs(int n){if(n<=2)return n;return climbStairs(n-1)+climbStairs(n-2);}
```

### Optimal -- O(N) time, O(1) space Fibonacci
```cpp
int climbStairs(int n){int a=1,b=2;for(int i=3;i<=n;i++){int c=a+b;a=b;b=c;}return n==1?1:b;}
```

## Problem 100: Min Cost Climbing Stairs
**LeetCode #746** | Pay cost[i] to step from stair i; start from 0 or 1.

### Brute Force -- O(2^N) recursion
```cpp
int minCostClimbingStairs_memo(vector<int>& c,int i,vector<int>& dp){
    if(i>=(int)c.size())return 0;
    if(dp[i]!=-1)return dp[i];
    return dp[i]=c[i]+min(minCostClimbingStairs_memo(c,i+1,dp),minCostClimbingStairs_memo(c,i+2,dp));
}
int minCostClimbingStairs(vector<int>& cost){int n=cost.size();vector<int>dp(n,-1);return min(minCostClimbingStairs_memo(cost,0,dp),minCostClimbingStairs_memo(cost,1,dp));}
```

### Optimal -- O(N) time, O(1) space
```cpp
int minCostClimbingStairs(vector<int>& cost) {
    int n=cost.size();
    for(int i=2;i<n;i++) cost[i]+=min(cost[i-1],cost[i-2]);
    return min(cost[n-1],cost[n-2]);
}
```

## Problem 101: House Robber
**LeetCode #198** | Max sum of non-adjacent elements.

### Brute Force -- O(2^N) try all subsets
### Optimal -- O(N) time, O(1) space
```cpp
int rob(vector<int>& nums){int a=0,b=0;for(int x:nums){int c=max(b,a+x);a=b;b=c;}return b;}
```

## Problem 103: Longest Palindromic Substring
**LeetCode #5**

### Brute Force -- O(N^3) check all substrings
```cpp
bool isPalin(string& s,int l,int r){while(l<r)if(s[l++]!=s[r--])return false;return true;}
string longestPalindrome_bf(string s){string res="";for(int i=0;i<(int)s.size();i++)for(int j=i;j<(int)s.size();j++)if(isPalin(s,i,j)&&j-i+1>(int)res.size())res=s.substr(i,j-i+1);return res;}
```

### Optimal -- O(N^2) expand around center
```cpp
string longestPalindrome(string s) {
    int n=s.size(),start=0,best=1;
    auto expand=[&](int l,int r){while(l>=0&&r<n&&s[l]==s[r]){if(r-l+1>best){best=r-l+1;start=l;}l--;r++;}};
    for(int i=0;i<n;i++){expand(i,i);expand(i,i+1);}
    return s.substr(start,best);
}
```

## Problem 104: Palindromic Substrings
**LeetCode #647** | Count all palindromic substrings.

### Brute Force -- O(N^3) check all substrings
### Optimal -- O(N^2) expand around center
```cpp
int countSubstrings(string s) {
    int n=s.size(),cnt=0;
    auto expand=[&](int l,int r){while(l>=0&&r<n&&s[l--]==s[r++])cnt++;};
    for(int i=0;i<n;i++){expand(i,i);expand(i,i+1);}
    return cnt;
}
```

## Problem 105: Decode Ways
**LeetCode #91** | Count decodings of digit string (1=A...26=Z).

### Brute Force -- O(2^N) recursion
### Optimal -- O(N) DP
```cpp
int numDecodings(string s) {
    int n=s.size(); vector<int> dp(n+1,0); dp[n]=1;
    for(int i=n-1;i>=0;i--){
        if(s[i]=='0'){dp[i]=0;continue;}
        dp[i]=dp[i+1];
        if(i+1<n&&(s[i]-'0')*10+(s[i+1]-'0')<=26) dp[i]+=dp[i+2];
    }
    return dp[0];
}
```

## Problem 106: Coin Change
**LeetCode #322** | Min coins to make amount (unlimited use).

### Brute Force -- O(S^C) recursion without memoization
### Optimal -- O(S * C) bottom-up DP
```cpp
int coinChange(vector<int>& coins, int amount) {
    vector<int> dp(amount+1,amount+1); dp[0]=0;
    for(int i=1;i<=amount;i++) for(int c:coins) if(c<=i) dp[i]=min(dp[i],1+dp[i-c]);
    return dp[amount]>amount?-1:dp[amount];
}
```

## Problem 107: Maximum Product Subarray
**LeetCode #152**

### Brute Force -- O(N^2) try all subarrays
### Optimal -- O(N) track both min and max product
```cpp
int maxProduct(vector<int>& nums) {
    int mx=nums[0],mn=nums[0],res=nums[0];
    for(int i=1;i<(int)nums.size();i++){
        if(nums[i]<0) swap(mx,mn);
        mx=max(nums[i],mx*nums[i]); mn=min(nums[i],mn*nums[i]);
        res=max(res,mx);
    }
    return res;
}
```

## Problem 108: Word Break
**LeetCode #139** | Can s be segmented into dictionary words?

### Brute Force -- O(2^N) recursion checking all splits
### Optimal -- O(N^2) DP with hash set
```cpp
bool wordBreak(string s, vector<string>& wordDict) {
    unordered_set<string> ws(wordDict.begin(),wordDict.end());
    int n=s.size(); vector<bool> dp(n+1,false); dp[0]=true;
    for(int i=1;i<=n;i++) for(int j=0;j<i;j++) if(dp[j]&&ws.count(s.substr(j,i-j))){dp[i]=true;break;}
    return dp[n];
}
```

## Problem 109: Longest Increasing Subsequence
**LeetCode #300**

### Brute Force -- O(N^2) DP
```cpp
int lengthOfLIS_dp(vector<int>& nums) {
    int n=nums.size(),res=1; vector<int> dp(n,1);
    for(int i=1;i<n;i++){for(int j=0;j<i;j++)if(nums[j]<nums[i])dp[i]=max(dp[i],dp[j]+1);res=max(res,dp[i]);}
    return res;
}
```

### Optimal -- O(N log N) patience sort (binary search on tails array)
```cpp
int lengthOfLIS(vector<int>& nums) {
    vector<int> tails;
    for(int x:nums){
        auto it=lower_bound(tails.begin(),tails.end(),x);
        if(it==tails.end()) tails.push_back(x);
        else *it=x;
    }
    return tails.size();
}
```

## Problem 110: Partition Equal Subset Sum
**LeetCode #416** | Can array be partitioned into two equal-sum subsets?

### Brute Force -- O(2^N) try all subsets
### Optimal -- O(N * S) 0/1 knapsack DP with bitset trick
```cpp
bool canPartition(vector<int>& nums) {
    int sum=accumulate(nums.begin(),nums.end(),0);
    if(sum%2) return false;
    int target=sum/2; vector<bool> dp(target+1,false); dp[0]=true;
    for(int x:nums) for(int j=target;j>=x;j--) dp[j]=dp[j]||dp[j-x];
    return dp[target];
}
```
